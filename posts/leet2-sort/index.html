    public int findKthLargest(int[] nums, int k) {
        PriorityQueue<Integer> pq = new PriorityQueue<>();
        for (int val : nums) {
            pq.add(val);
            if (pq.size() > k)  // з»ҙжҠӨе Ҷзҡ„еӨ§е°Ҹдёә K
                pq.poll();
        }
        return pq.peek();
    }
//ж—¶й—ҙеӨҚжқӮеәҰ O(NlogK)пјҢз©әй—ҙеӨҚжқӮеәҰ O(K)гҖӮ

public int findKthLargest3(int[] nums, int k) {
        int left = 0, right = nums.length - 1;
        k = nums.length - k;
        while (left < right) {
            int j = partition(nums, left, right);
            if (j == k)
                break;
            else if (j < k) {
                left = j + 1;
            } else {
                right = j - 1;
            }
        }
        return nums[k];
    }

    public int partition(int num[], int l, int h) {
        int i = l, j = h + 1;
        while (true) {
            while (num[++i] < num[l] && i < h) ;
            while (num[--j] > num[l] && j > l) ;
            if (i >= j) {
                break;
            }
            swap(num, i, j);
        }
        swap(num, l, j);
        return j;
    }

    public void swap(int[] a, int i, int j) {
        int t = a[i];
        a[i] = a[j];
        a[j] = t;
    }
//ж—¶й—ҙеӨҚжқӮеәҰ O(N)пјҢз©әй—ҙеӨҚжқӮеәҰ O(1)

  1. Top K Frequent Elements (Medium)

Leetcode / еҠӣжүЈ

Given [1,1,1,2,2,3] and k = 2, return [1,2].

    public List<Integer> topKFrequent(int[] nums, int k) {
        List<Integer> res = new ArrayList();
        // дҪҝз”Ёеӯ—е…ёпјҢз»ҹи®ЎжҜҸдёӘе…ғзҙ еҮәзҺ°зҡ„ж¬Ўж•°пјҢе…ғзҙ дёәй”®пјҢе…ғзҙ еҮәзҺ°зҡ„ж¬Ўж•°дёәеҖј
        HashMap<Integer, Integer> map = new HashMap();
        for (int num : nums) {
            if (map.containsKey(num)) {
                map.put(num, map.get(num) + 1);
            } else {
                map.put(num, 1);
            }
        }

        //жЎ¶жҺ’еәҸ
        //е°Ҷйў‘зҺҮдҪңдёәж•°з»„дёӢж ҮпјҢеҜ№дәҺеҮәзҺ°йў‘зҺҮдёҚеҗҢзҡ„ж•°еӯ—йӣҶеҗҲпјҢеӯҳе…ҘеҜ№еә”зҡ„ж•°з»„дёӢж Ү
        List<Integer>[] list = new List[nums.length + 1];
        for (int key : map.keySet()) {
            // иҺ·еҸ–еҮәзҺ°зҡ„ж¬Ўж•°дҪңдёәдёӢж Ү
            int i = map.get(key);
            if (list[i] == null) {
                list[i] = new ArrayList();
            }
            list[i].add(key);
        }

        // еҖ’еәҸйҒҚеҺҶж•°з»„иҺ·еҸ–еҮәзҺ°йЎәеәҸд»ҺеӨ§еҲ°е°Ҹзҡ„жҺ’еҲ—
        for (int i = list.length - 1; i >= 0 && res.size() < k; i--) {
            if (list[i] == null) continue;
            res.addAll(list[i]);
        }
        return res;
    }

  1. Sort Characters By Frequency (Medium)

Leetcode / еҠӣжүЈ

Input:
"tree"

Output:
"eert"

Explanation:
'e' appears twice while 'r' and 't' both appear once.
So 'e' must appear before both 'r' and 't'. Therefore "eetr" is also a valid answer.

жЎ¶жҺ’еәҸ

class Solution {
    public String frequencySort(String s) {
        HashMap<Character, Integer> hashMap = new HashMap<>();
        char[] str = s.toCharArray();
        for (char c : str) {
            //System.out.print(c + " ");
            if (hashMap.containsKey(c)) {
                hashMap.put(c, hashMap.get(c) + 1);
            } else {
                hashMap.put(c, 1);
            }//hashMapз»ҹи®Ўеӯ—з¬ҰеҮәзҺ°дёӘж•°
        }//Keyдёәеӯ—з¬ҰValueдёәеӯ—з¬Ұ
        List<Character>[] bucket = new List[str.length + 1];
        for (Character character : hashMap.keySet()) {
            int index = hashMap.get(character);
//            System.out.print("key--"+character+" ");
//            System.out.println("value--"+index+" ");
            if (bucket[index] == null) {
                bucket[index] = new ArrayList<Character>();
            }
            bucket[index].add(character);
            //System.out.println(bucket[1]);
        }
        //еҲӣе»әListжҢүйЎәеәҸеӯҳеӮЁеӯ—з¬Ұ
        StringBuilder st = new StringBuilder();
        for (int i = str.length; i >= 0; i--) {
            if (bucket[i] == null) continue;
            for (Character character : bucket[i]) {
                //иҝҷиҫ№жіЁж„Ҹеӯ—з¬ҰжҢүеҮәзҺ°ж¬Ўж•°append
                for (int j = 0; j < i; j++) {
                    st.append(character);
                }
            }
        }
        return new String(st);
    }
}
