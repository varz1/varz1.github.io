Map

Go语言中提供的映射关系容器为map，其内部使用散列表（hash）实现。map是一种无序的基于key-value的数据结构，Go语言中的map是引用类型，必须初始化才能使用。

使用

eg1.

scoreMap :=make(map[string]int ,8)
scoreMap["小红"]=90
scoreMap["小蓝"]=91
fmt.Println(scoreMap)

输出

map[小红:90 小蓝:91]

eg2.

scoreMap := map[string]int{
	"小红" : 90,
	"小蓝" : 91,
}
fmt.Println(scoreMap)

查找

v,ok := scoreMap["小红"]
	if ok{
		fmt.Printf("对应值为%v\n",v)
	} else {
		fmt.Println("查无此人")
	}

遍历

for k,v := range scoreMap{
	fmt.Println(k,v)
}

元素为map类型的切片

	sliceMap1 := make([]map[string]string, 3)
	sliceMap1[0] = make(map[string]string, 10)
	sliceMap1[0]["name"] = "陈哈哈"
	sliceMap1[0]["age"] = "18"
	for i,v:=range sliceMap1{
		fmt.Printf("index:%v,value:%v",i,v)
	}
	fmt.Println(sliceMap1)

元素为切片类型的map

	var mapSlice = make(map[string][]string, 3)
	fmt.Println(mapSlice)
	key := "中国"
	value, ok := mapSlice[key]
	if !ok {
		value = make([]string, 0, 2)
	}
	value = append(value, "北京", "上海")
	mapSlice[key] = value
	fmt.Println(mapSlice)
