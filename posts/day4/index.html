import "fmt"

func main() {
	if age := 18; age >= 18 {
		fmt.Println("wb欢迎你")
	} else {
		fmt.Println("该写作业啦")
	}

	//for1
	fmt.Println("for1")
	for i := 0; i < 5; i++ {
		fmt.Printf("第%d次循环\n", i)
	}
	//for2
	fmt.Println("for2")
	i := 3
	for ; i < 5; i++ {
		fmt.Printf("第%d次循环\n", i)
	}
	//for2
	fmt.Println("for3")
	for i > 3 {
		fmt.Println("1")
		i--
	}
	//for range
	fmt.Println("for range")
	s := "halo 你好"
	for i, v := range s {
		fmt.Printf("%d %c\n",i, v)
	}
	//switch case
	//只能有一个default分支
	color := 1
	switch color {
	case 1:
		fmt.Println("red")
	case 2:
		fmt.Println("green")
	case 3:
		fmt.Println("yello")
	default:
		fmt.Println("无效输入")
	}
	//goto
	for i := 0; i < 10; i++ {
		for j := 0; j < 10; j++ {
			if j == 2 {
				// 设置退出标签
				goto breakTag
			}
			fmt.Printf("%v-%v\n", i, j)
		}
	}
	return
	//break标签
breakTag:
	fmt.Println("结束for循环")
	BREAKDEMO1:
		for i := 0; i < 10; i++ {
			for j := 0; j < 10; j++ {
				if j == 2 {
					break BREAKDEMO1
				}
				fmt.Printf("%v-%v\n", i, j)
			}
		}
		fmt.Println("...")
forloop1:
	for i := 0; i < 5; i++ {
		// forloop2:
		for j := 0; j < 5; j++ {
			if i == 2 && j == 2 {
				continue forloop1
			}
			fmt.Printf("%v-%v\n", i, j)
		}
	}
}
