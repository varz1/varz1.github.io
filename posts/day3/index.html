布尔值

  1. bool只有true和false
  2. 默认值为false

   	fmt.Println(strings.HasPrefix(d,"ha"))

   	//判断位置
   	fmt.Println("判断位置")
   	fmt.Println(strings.Index(d,"o"))//第一次出现的位置
   	fmt.Println(strings.LastIndex(d,"o"))//最后一次出现的位置
   	//拼接
   	fmt.Println("拼接")
   	fmt.Println(strings.Join(ret,"+"))

## 字符串操作

  • 字符串编码方式不同中文无法使用字符数组需要转换使用

测试

   package main
   
   import (
   	"fmt"
   	"strings"
   )
   
   func main() {
   	s1 := 795
   	fmt.Printf("十进制%d\n", s1)
   	fmt.Printf("二进制%b\n", s1)
   	fmt.Printf("八进制%o\n", s1)
   	fmt.Printf("十六进制%x\n", s1)
   	fmt.Printf("查看类型%T\n", s1)
   
   	s2 := 1.23456798
   	fmt.Printf("默认浮点类型为%T\n", s2)
   	s3 := float32(1.23456)
   	//不同类型无法赋值s3 = s2
   	fmt.Printf("s3浮点类型为%T\n", s3)
   	c1 := 1 + 2i
   	c2 := 1 + 3i
   	c3 := c1 + c2
   	fmt.Printf("c1+c2=%v", c3) //%v 万能占位符
   
   	//多行字符串
   	s4 := `
   		1
   		2
   123
   	`
   	fmt.Println(s4)
   
   	//字符串相关操作
   	//求长度
   	fmt.Println(len(s4))
   	//字符串拼接
   	a := "halo"
   	b := "world"
   	fmt.Println(a+b)
   	d := fmt.Sprintf("%s%s",a,b)//返回值给d
   	fmt.Println(d)
   	//字符串分割
   	fmt.Println("字符串分割//把某个字符串用空格代替")
   	ret := strings.Split(d,"o")
   	fmt.Println(ret)
   	//判断包含
   	fmt.Println("判断包含")
   	fmt.Println(strings.Contains(d,"halo"))
   	//前缀后缀判断
   	fmt.Println("前缀后缀判断")
   	fmt.Println(strings.HasPrefix(d,"ha"))
   	fmt.Println(strings.HasSuffix(d,"ha"))
   	//判断位置
   	fmt.Println("判断位置")
   	fmt.Println(strings.Index(d,"o"))//第一次出现的位置
   	fmt.Println(strings.LastIndex(d,"o"))//最后一次出现的位置
   	//拼接
   	fmt.Println("拼接")
   	fmt.Println(strings.Join(ret,"+"))
   	//字符串操作
   
   	//分开打印
   	n := len(d)
   	fmt.Println(n)
   	for _, c := range d{
   		fmt.Printf("%c\n",c)
	}
   }
